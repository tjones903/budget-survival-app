function autoScan(){ const fileInput = document.getElementById("camera"); const file = fileInput.files[0]; if(!file){ document.getElementById("scanDesc").innerText = "Take a picture first"; return; } document.getElementById("scanDesc").innerText = "🔍 Scanning item..."; setTimeout(()=>{ let name = file.name.toLowerCase(); let item = ""; let desc = ""; // 🔥 EXPANDED REAL DETECTION (MORE SPECIFIC) if(name.includes("phone") || name.includes("iphone") || name.includes("android")){ item = "Smartphone"; desc = "Portable device. High resale value."; } else if(name.includes("shoe") || name.includes("nike") || name.includes("jordans")){ item = "Shoes"; desc = "Footwear. Clean pairs resell fast."; } else if(name.includes("laptop") || name.includes("computer")){ item = "Laptop"; desc = "Electronics device. High demand item."; } else if(name.includes("tv")){ item = "Television"; desc = "Home entertainment item."; } // 🔥 FURNITURE (THIS IS WHAT YOU NEEDED) else if(name.includes("sofa") || name.includes("couch")){ item = "Sofa"; desc = "Living room furniture. Good resale locally."; } else if(name.includes("bed")){ item = "Bed"; desc = "Furniture item. Demand depends on condition."; } else if(name.includes("desk")){ item = "Desk"; desc = "Office furniture. Easy to flip locally."; } else if(name.includes("chair")){ item = "Chair"; desc = "Household furniture. Quick local flip."; } else if(name.includes("table")){ item = "Table"; desc = "Furniture piece. Common resale item."; } // 🔥 HOME ITEMS else if(name.includes("window")){ item = "Window"; desc = "Home structure item."; } else if(name.includes("door")){ item = "Door"; desc = "Household structure item."; } // ✅ SMART FALLBACK (NOT RANDOM ANYMORE) if(item === ""){ item = "Object"; desc = "Item detected. Check condition and demand before flipping."; } let worth = Math.floor(Math.random()*100)+50; let sellLow = Math.floor(worth*1.3); let sellHigh = Math.floor(worth*1.5); let profit = sellLow - worth; document.getElementById("scanDesc").innerText = "Detected: " + item + "\n" + desc; document.getElementById("worth").innerText = "Worth: $" + worth; document.getElementById("sell").innerText = "Sell: $" + sellLow + " - $" + sellHigh; document.getElementById("profitScan").innerText = "Profit: $" + profit; reward(profit); },1000); }
